String foreignRelationToPgsql() {
  var out = '';

  if (foreignKeys.isEmpty) return out;

  // Foreign keys
  for (var key in foreignKeys) {
    out += key.toPgSql(tableName: name);
  }

  out += '\n';

  return out;
}
